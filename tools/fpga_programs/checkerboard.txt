start:
	ctrl nop
	ctrl addi R1 ZERO 0 # Rows = 0

whiteblack:
	ctrl nop
	ctrl addi R2 ZERO 0 # Cols = 0

wb_white_start:
	ctrl nop
	ctrl addi VADDR VADDR 1
	ctrl addi VDATA ZERO 255
	ctrl addi R2 R2 1 # Increment cols
	ctrl beq wb_white_start
	ctrl slti ZERO R2 160
	ctrl nop

	ctrl addi R2 ZERO 0 # Cols = 0

wb_black_start:
	ctrl nop
	ctrl addi VADDR VADDR 1
	ctrl addi VDATA ZERO 0
	ctrl addi R2 R2 1 # Increment cols
	ctrl beq wb_black_start
	ctrl slti ZERO R2 160
	ctrl nop

	ctrl addi R1 R1 1 # Increment rows
	ctrl beq whiteblack
	ctrl slti ZERO R1 120
	ctrl nop

	ctrl addi R1 ZERO 0 # Rows = 0

blackwhite:
	ctrl nop
	ctrl addi R2 ZERO 0 # Cols = 0

bw_black_start:
	ctrl nop
	ctrl addi VADDR VADDR 1
	ctrl addi VDATA ZERO 0
	ctrl addi R2 R2 1 # Increment cols
	ctrl beq bw_black_start
	ctrl slti ZERO R2 160
	ctrl nop

	ctrl addi R2 ZERO 0 # Cols = 0

bw_white_start:
	ctrl nop
	ctrl addi VADDR VADDR 1
	ctrl addi VDATA ZERO 255
	ctrl addi R2 R2 1 # Increment cols
	ctrl beq bw_white_start
	ctrl slti ZERO R2 160
	ctrl nop

	ctrl addi R1 R1 1 # Increment rows
	ctrl beq blackwhite
	ctrl slti ZERO R1 120
	ctrl nop

	ctrl jump start
