pixel_loop:

    ctrl lw VDATA VADDR
    ctrl nop
    ctrl addi VADDR VADDR 1

    ctrl beq infinite_loop
    ctrl slt ZERO R1 VADDR
    ctrl nop

    ctrl jump pixel_loop

infinite_loop:

    ctrl nop
